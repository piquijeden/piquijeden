  <script>

    const canvas = document.getElementById('game');

    const ctx = canvas.getContext('2d');

    const scoreEl = document.getElementById('score');

    const livesEl = document.getElementById('lives');

    const bgm = document.getElementById('bgm');


    const W = 800;

    const H = 550;


    let score = 0;

    let lives = 3;

    let paused = false;

    let gameOver = false;


    const keys = { left: false, right: false };


    const janusz = {

      x: 360,

      y: H - 220,

      speed: 7,

      basketOffsetX: 55,

      basketOffsetY: 110,

      basketWidth: 150,

      basketHeight: 38

    };


    const products = [

      { name: 'Chleb', color: '#d9a55a' },

      { name: 'Mleko', color: '#bfe3ff' },

      { name: 'Ser', color: '#ffd95e' },

      { name: 'Cebula, color: '#f3e04f' },

      { name: 'Czteropak’, color: '#ff9b3d' },

      { name: 'Parówki', color: '#de6b6b' },

      { name: 'Wódka', color: '#7ccf6a' },

      { name: 'Masło', color: '#fff2a8' }

    ];


    let fallSpeed = 4.8;

    let product = null;


    const pauseButtons = {

      continue: { x: 290, y: 255, w: 220, h: 50 },

      restart: { x: 290, y: 330, w: 220, h: 50 },

      slider: { x: 250, y: 420, w: 300, h: 8, handleR: 12 }

    };


    let volume = 0.5;

    bgm.volume = volume;


    function updateHud() {

      scoreEl.textContent = `Punkty: ${score}`;

      livesEl.textContent = `Życia: ${lives}`;

    }


    function spawnProduct() {

      const item = products[Math.floor(Math.random() * products.length)];

      product = {

        x: Math.random() * (W - 52),

        y: 120,

        size: 52,

        name: item.name,

        color: item.color

      };

    }


    function getBasketRect() {

      return {

        x: janusz.x + janusz.basketOffsetX,

        y: janusz.y + janusz.basketOffsetY,

        w: janusz.basketWidth,

        h: janusz.basketHeight

      };

    }


    function intersects(a, b) {

      return a.x < b.x + b.w && a.x + a.w > b.x && a.y < b.y + b.h && a.y + a.h > b.y;

    }


    function resetGame() {

      score = 0;

      lives = 3;

      paused = false;

      gameOver = false;

      fallSpeed = 4.8;

      janusz.x = 360;

      spawnProduct();

      updateHud();

    }


    function update() {

      if (!paused && !gameOver) {

        if (keys.left) janusz.x -= janusz.speed;

        if (keys.right) janusz.x += janusz.speed;


        const minX = -85;

        const maxX = W - janusz.basketWidth - 55;

        janusz.x = Math.max(minX, Math.min(maxX, janusz.x));


        if (product) {

          product.y += fallSpeed;


          const p = { x: product.x, y: product.y, w: product.size, h: product.size };

          const b = getBasketRect();


          if (intersects(p, b)) {

            score++;

            if (score % 5 === 0) fallSpeed += 0.35;

            spawnProduct();

            updateHud();

          } else if (product.y > H) {

            lives--;

            updateHud();

            if (lives <= 0) {

              gameOver = true;

            } else {

              spawnProduct();

            }

          }

        }

      }

    }


    function drawBackground() {

      ctx.fillStyle = '#f5f1e8';

      ctx.fillRect(0, 0, W, H);


      ctx.fillStyle = '#d8d8d8';

      ctx.fillRect(0, H - 120, W, 120);


      const shelfW = W / 4;

      const shelves = [

        { x: 40, color: '#ffe14d', text: 'PROMOCJE' },

        { x: 80 + shelfW, color: '#ff5757', text: 'NABIAŁ' },

        { x: 120 + shelfW * 2, color: '#7fd36b', text: ‚ALKOHOLE }

      ];


      ctx.lineWidth = 3;

      ctx.strokeStyle = '#333';

      ctx.font = 'bold 20px Arial';

      ctx.textAlign = 'center';

      ctx.textBaseline = 'middle';


      for (const s of shelves) {

        ctx.fillStyle = s.color;

        ctx.fillRect(s.x, 70, shelfW, 100);

        ctx.strokeRect(s.x, 70, shelfW, 100);

        ctx.fillStyle = '#111';

        ctx.fillText(s.text, s.x + shelfW / 2, 120);

      }

    }


    function drawJanusz() {

      const x = janusz.x;

      const y = janusz.y;


      ctx.strokeStyle = 'black';

      ctx.lineWidth = 5;

      ctx.beginPath();

      ctx.moveTo(x - 20, y + 150);

      ctx.lineTo(x - 25, y + 190);

      ctx.moveTo(x + 20, y + 150);

      ctx.lineTo(x + 25, y + 190);

      ctx.stroke();


      ctx.fillStyle = '#4d83ff';

      ctx.lineWidth = 2;

      ctx.fillRect(x - 40, y + 70, 80, 80);

      ctx.strokeRect(x - 40, y + 70, 80, 80);


      ctx.lineWidth = 5;

      ctx.beginPath();

      ctx.moveTo(x - 40, y + 85);

      ctx.lineTo(x - 80, y + 110);

      ctx.moveTo(x + 40, y + 85);

      ctx.lineTo(x + 70, y + 110);

      ctx.stroke();


      ctx.beginPath();

      ctx.lineWidth = 2;

      ctx.fillStyle = '#f0c9a0';

      ctx.arc(x, y + 35, 35, 0, Math.PI * 2);

      ctx.fill();

      ctx.stroke();


      ctx.fillStyle = 'black';

      ctx.beginPath();

      ctx.arc(x - 14, y + 26, 4, 0, Math.PI * 2);

      ctx.arc(x + 14, y + 26, 4, 0, Math.PI * 2);

      ctx.fill();


      ctx.fillStyle = '#d8a98d';

      ctx.beginPath();

      ctx.ellipse(x + 18, y + 37, 13, 9, 0, 0, Math.PI * 2);

      ctx.fill();

      ctx.stroke();


      ctx.beginPath();

      ctx.arc(x - 6, y + 46, 16, 0, Math.PI, false);

      ctx.stroke();


      const bx = x + janusz.basketOffsetX;

      const by = y + janusz.basketOffsetY;

      const bw = janusz.basketWidth;

      const bh = janusz.basketHeight;


      ctx.fillStyle = '#9c6736';

      ctx.fillRect(bx, by, bw, bh);

      ctx.strokeRect(bx, by, bw, bh);


      ctx.beginPath();

      ctx.arc(bx + bw / 2, by + 2, bw / 2 - 18, Math.PI, 0, false);

      ctx.stroke();


      for (let i = 1; i < 6; i++) {

        const lx = bx + i * (bw / 6);

        ctx.beginPath();

        ctx.moveTo(lx, by + 3);

        ctx.lineTo(lx, by + bh - 3);

        ctx.stroke();

      }


      ctx.beginPath();

      ctx.moveTo(bx + 5, by + 12);

      ctx.lineTo(bx + bw - 5, by + 12);

      ctx.moveTo(bx + 5, by + 25);

      ctx.lineTo(bx + bw - 5, by + 25);

      ctx.stroke();

    }


    function drawProduct() {

      if (!product) return;


      ctx.fillStyle = product.color;

      ctx.fillRect(product.x, product.y, product.size, product.size);

      ctx.strokeStyle = 'black';

      ctx.lineWidth = 2;

      ctx.strokeRect(product.x, product.y, product.size, product.size);


      ctx.fillStyle = '#111';

      ctx.font = 'bold 9px Arial';

      ctx.textAlign = 'center';

      ctx.textBaseline = 'middle';

      wrapText(product.name, product.x + product.size / 2, product.y + product.size / 2, product.size - 8, 10);

    }


    function wrapText(text, x, y, maxWidth, lineHeight) {

      const words = text.split(' ');

      let line = '';

      let lines = [];


      for (let n = 0; n < words.length; n++) {

        const testLine = line + words[n] + ' ';

        const testWidth = ctx.measureText(testLine).width;

        if (testWidth > maxWidth && n > 0) {

          lines.push(line.trim());

          line = words[n] + ' ';

        } else {

          line = testLine;

        }

      }

      lines.push(line.trim());


      const startY = y - ((lines.length - 1) * lineHeight) / 2;

      lines.forEach((ln, i) => ctx.fillText(ln, x, startY + i * lineHeight));

    }


    function drawGameOver() {

      ctx.fillStyle = 'white';

      ctx.strokeStyle = 'black';

      ctx.lineWidth = 3;

      ctx.fillRect(W / 2 - 220, H / 2 - 100, 440, 200);

      ctx.strokeRect(W / 2 - 220, H / 2 - 100, 440, 200);


      ctx.fillStyle = 'red';

      ctx.font = 'bold 28px Arial';

      ctx.textAlign = 'center';

      ctx.fillText('KONIEC GRY', W / 2, H / 2 - 35);


      ctx.fillStyle = 'black';

      ctx.font = 'bold 20px Arial';

      ctx.fillText(`Twój wynik: ${score}`, W / 2, H / 2 + 5);


      ctx.font = '16px Arial';

      ctx.fillText('Naciśnij R, aby zagrać ponownie', W / 2, H / 2 + 45);

    }


    function drawPauseMenu() {

      ctx.fillStyle = 'rgba(0, 0, 0, 0.45)';

      ctx.fillRect(0, 0, W, H);


      ctx.fillStyle = 'white';

      ctx.strokeStyle = 'black';

      ctx.lineWidth = 3;

      ctx.fillRect(W / 2 - 185, H / 2 - 140, 370, 340);

      ctx.strokeRect(W / 2 - 185, H / 2 - 140, 370, 340);


      ctx.fillStyle = 'black';

      ctx.font = 'bold 28px Arial';

      ctx.textAlign = 'center';

      ctx.fillText('PAUZA', W / 2, H / 2 - 90);


      drawButton(pauseButtons.continue, '#7fd36b', 'Kontynuuj');

      drawButton(pauseButtons.restart, '#ff8a8a', 'Restart');


      ctx.font = 'bold 12px Arial';

      ctx.fillStyle = 'black';

      ctx.fillText(`Głośność: ${Math.round(volume * 100)}%`, W / 2, H / 2 + 140);


      const s = pauseButtons.slider;

      ctx.strokeStyle = '#666';

      ctx.lineWidth = 6;

      ctx.beginPath();

      ctx.moveTo(s.x, s.y);

      ctx.lineTo(s.x + s.w, s.y);

      ctx.stroke();


      const handleX = s.x + s.w * volume;

      ctx.fillStyle = '#2f6fff';

      ctx.beginPath();

      ctx.arc(handleX, s.y, s.handleR, 0, Math.PI * 2);

      ctx.fill();

      ctx.strokeStyle = '#111';

      ctx.lineWidth = 2;

      ctx.stroke();

    }


    function drawButton(btn, color, text) {

      ctx.fillStyle = color;

      ctx.fillRect(btn.x, btn.y, btn.w, btn.h);

      ctx.strokeStyle = 'black';

      ctx.lineWidth = 2;

      ctx.strokeRect(btn.x, btn.y, btn.w, btn.h);

      ctx.fillStyle = 'black';

      ctx.font = 'bold 18px Arial';

      ctx.textAlign = 'center';

      ctx.textBaseline = 'middle';

      ctx.fillText(text, btn.x + btn.w / 2, btn.y + btn.h / 2);

    }


    function draw() {

      drawBackground();

      drawJanusz();

      drawProduct();

      if (gameOver) drawGameOver();

      if (paused && !gameOver) drawPauseMenu();

    }


    function loop() {

      update();

      draw();

      requestAnimationFrame(loop);

    }


    function inside(mx, my, rect) {

      return mx >= rect.x && mx <= rect.x + rect.w && my >= rect.y && my <= rect.y + rect.h;

    }


    function tryStartMusic() {

      bgm.play().catch(() => {});

    }


    window.addEventListener('keydown', (e) => {

      if (e.key === 'ArrowLeft') keys.left = true;

      if (e.key === 'ArrowRight') keys.right = true;


      if (e.key === 'Escape' && !gameOver) {

        paused = !paused;

      }


      if ((e.key === 'r' || e.key === 'R') && gameOver) {

        resetGame();

      }


      tryStartMusic();

    });


    window.addEventListener('keyup', (e) => {

      if (e.key === 'ArrowLeft') keys.left = false;

      if (e.key === 'ArrowRight') keys.right = false;

    });


    let draggingSlider = false;


    canvas.addEventListener('mousedown', (e) => {

      const rect = canvas.getBoundingClientRect();

      const mx = (e.clientX - rect.left) * (canvas.width / rect.width);

      const my = (e.clientY - rect.top) * (canvas.height / rect.height);


      tryStartMusic();


      if (!paused) return;


      if (inside(mx, my, pauseButtons.continue)) {

        paused = false;

        return;

      }


      if (inside(mx, my, pauseButtons.restart)) {

        resetGame();

        return;

      }


      const s = pauseButtons.slider;

      const handleX = s.x + s.w * volume;

      const dx = mx - handleX;

      const dy = my - s.y;

      if (Math.sqrt(dx * dx + dy * dy) <= s.handleR + 6 || (mx >= s.x && mx <= s.x + s.w && my >= s.y - 12 && my <= s.y + 12)) {

        draggingSlider = true;

        setVolumeFromMouse(mx);

      }

    });


    window.addEventListener('mousemove', (e) => {

      if (!draggingSlider) return;

      const rect = canvas.getBoundingClientRect();

      const mx = (e.clientX - rect.left) * (canvas.width / rect.width);

      setVolumeFromMouse(mx);

    });


    window.addEventListener('mouseup', () => {

      draggingSlider = false;

    });


    function setVolumeFromMouse(mx) {

      const s = pauseButtons.slider;

      volume = (mx - s.x) / s.w;

      volume = Math.max(0, Math.min(1, volume));

      bgm.volume = volume;

    }


    spawnProduct();

    updateHud();

    draw();

    loop();

  </script>
